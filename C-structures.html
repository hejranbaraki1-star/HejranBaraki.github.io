📝 Exercise 8: Structures

Description: Creating and using structures (struct) in C.

Code:

#include <stdio.h>
#include <string.h>

// Define structure
struct Student {
    char name[50];
    int age;
    float gpa;
};

int main() {
    struct Student student1;
    
    strcpy(student1.name, "John Doe");
    student1.age = 20;
    student1.gpa = 3.8;
    
    printf("Name: %s\n", student1.name);
    printf("Age: %d\n", student1.age);
    printf("GPA: %.2f\n", student1.gpa);
    return 0;
}
