📝 Exercise 11: Sorting Algorithms

Description: Implementing Bubble Sort and other sorting algorithms in C.

Bubble Sort:

#include <stdio.h>

void bubbleSort(int arr[], int n) {
    for(int i = 0; i < n-1; i++) {
        for(int j = 0; j < n-i-1; j++) {
            if(arr[j] > arr[j+1]) {
                // Swap
                int temp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = temp;
            }
        }
    }
}

int main() {
    int numbers[] = {64, 34, 25, 12, 22, 11, 90};
    int n = 7;
    
    bubbleSort(numbers, n);
    
    printf("Sorted array: ");
    for(int i = 0; i < n; i++) {
        printf("%d ", numbers[i]);
    }
    return 0;
}
